70static void ConvertirAD(void *param)
71{
72 while (1) {
73 // Wait for the notification from the FuncTimerA ISR to perform the task.
74 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
75
76 uint16_t lectura;
77 // Read the analog signal from channel CH1 and store the digital value in the 'lectura' variable.
78 AnalogInputReadSingle(CH1, &lectura);

80 // Convert the digital value to a string and send it to the PC via UART.
