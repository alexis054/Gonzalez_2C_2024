52static void MedirDistancia(void *pvParameter)
53{
54 while(1)
55 {
56 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
57 if (Activate==true)
58 {
59 distancia = HcSr04ReadDistanceInCentimeters();
60
61 if (distancia < DISTANCIA_MENOR)
62 {
63 LedsOffAll();
64 }
65 if (distancia > DISTANCIA_MENOR && distancia < DISTANCIA_MEDIA)
66 {
67 LedOn(LED_1);
68 LedOff(LED_2);
69 LedOff(LED_3);
70 }
71
72 if (distancia > DISTANCIA_MEDIA && distancia < DISTANCIA_MAXIMA)
73 {
74 LedOn(LED_1);
75 LedOn(LED_2);
76 LedOff(LED_3);
77 }
78 if (distancia > DISTANCIA_MAXIMA)
79 {
80 LedOn(LED_1);
81 LedOn(LED_2);
82 LedOn(LED_3);
83 }
84
85 if (Hold == false)
86 {
87 LcdItsE0803Write(distancia);
88 UartSendString(UART_PC, (char *)UartItoa(distancia, 10));
89 UartSendString(UART_PC, " cm ");
90 UartSendString(UART_PC, "\r\n" );
91 distancia_hold = distancia;
92 }
93 else
94 {
95 UartSendString(UART_PC, (char *)UartItoa(distancia_hold, 10));
96 UartSendString(UART_PC, " cm ");
97 UartSendString(UART_PC, "\r\n" );
98 }
99
100
101 }
102 }
103}
